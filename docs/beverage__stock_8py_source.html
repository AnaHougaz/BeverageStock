71class CalculadoraCusto:
72 """
73 @class CalculadoraCusto
74 @brief Responsável pelos cálculos de logística e gestão de estoque
75
76 Implementa fórmulas clássicas de gestão de inventário como EOQ
77 (Economic Order Quantity) e cálculos de estoque de segurança.
78 """
79
80 @staticmethod
81 def calcular_estoque_seguranca(desvio_padrao: float, dias_seguranca: int,
82 demanda_media_diaria: float) -> int:
83 """
84 @brief Calcula a quantidade ideal de Estoque de Segurança
85
86 Baseado na fórmula: Fator de Serviço × Desvio Padrão × √(Lead Time)
87
88 @param desvio_padrao Variação na demanda ou no prazo de entrega
89 @param dias_seguranca Fator de serviço em dias (Ex: 7 dias)
90 @param demanda_media_diaria Consumo médio diário do produto
91 @return A quantidade mínima que deve estar sempre em estoque
92
93 @note Um valor típico de dias_seguranca é 7 para produtos de alta rotação
94 """
95 estoque_seguranca = desvio_padrao * dias_seguranca * demanda_media_diaria
96 return math.ceil(estoque_seguranca)
97
98 @staticmethod
99 def determinar_ponto_pedido(lead_time_dias: int, demanda_media_diaria: float,
100 estoque_seguranca: int) -> int:
101 """
102 @brief Determina o Ponto de Pedido (Reorder Point - ROP)
103
104 Indica quando um novo pedido deve ser feito para evitar a falta de estoque.
105 Fórmula: ROP = (Lead Time × Demanda Média) + Estoque de Segurança
106
107 @param lead_time_dias O tempo de entrega do fornecedor em dias
108 @param demanda_media_diaria O consumo médio diário do produto
109 @param estoque_seguranca O valor mínimo calculado
110 @return A quantidade total que deve acionar o novo pedido
111
112 @warning Se o estoque atual cair abaixo deste valor, faça um pedido imediatamente!
113 """
114 ponto_pedido = (lead_time_dias * demanda_media_diaria) + estoque_seguranca
115 return math.ceil(ponto_pedido)
116
117 @staticmethod
118 def calcular_lote_economico(demanda_anual: float, custo_pedido: float,
119 custo_manutencao: float) -> int:
120 """
121 @brief Calcula o Lote Econômico de Compra (EOQ - Economic Order Quantity)
122
123 Determina a quantidade ideal a ser pedida que minimiza o custo total
124 de estoque (custos de pedido + custos de manutenção).
125
126 Fórmula EOQ: √((2 × Demanda Anual × Custo por Pedido) / Custo de Manutenção)
127
128 @param demanda_anual Demanda total prevista para o ano
129 @param custo_pedido Custo fixo de fazer um pedido
130 @param custo_manutencao Custo de manter uma unidade em estoque por ano
131 @return A quantidade ótima a ser pedida
132
133 @see https://en.wikipedia.org/wiki/Economic_order_quantity
134 """
135 if custo_manutencao == 0:
136 raise ValueError("Custo de manutenção não pode ser zero")
137
138 eoq = math.sqrt((2 * demanda_anual * custo_pedido) / custo_manutencao)
139 return math.ceil(eoq)
